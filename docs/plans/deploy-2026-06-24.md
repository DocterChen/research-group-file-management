3. **安全基线**：完成认证、授权、输入验证、密钥管理的最小安全检查
4. **可运维性**：支持服务启停、日志查看、配置更新、版本回滚

### 阶段 1：环境准备（预计 30 分钟）

#### 1.1 服务器要求
- **操作系统**：Ubuntu 20.04+ / CentOS 8+ / Debian 11+
- **CPU/内存**：2 核 / 2GB+（小规模使用，100 用户以内）
- **磁盘空间**：20GB+（含 Docker 镜像、数据文件、日志）
- **网络**：公网 IP 或内网 IP + 域名解析

#### 1.2 安装依赖
```bash
# 更新系统包
sudo apt update && sudo apt upgrade -y

# 安装 Docker
curl -fsSL https://get.docker.com | sh
sudo usermod -aG docker $USER
newgrp docker

# 安装 Docker Compose
sudo curl -L "https://github.com/docker/compose/releases/latest/download/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
sudo chmod +x /usr/local/bin/docker-compose

# 验证安装
docker --version
docker-compose --version
```

#### 1.3 防火墙配置
```bash
# 开放 HTTP/HTTPS 端口
sudo ufw allow 80/tcp
sudo ufw allow 443/tcp
sudo ufw enable
```

### 阶段 2：项目部署文件创建（预计 1 小时）

#### 2.1 创建 `Dockerfile`
```dockerfile
FROM python:3.11-slim

# 设置工作目录
WORKDIR /app

# 复制项目文件
COPY pyproject.toml ./
COPY src/ ./src/

# 安装依赖
RUN pip install --no-cache-dir -e . && \
    pip install openpyxl requests PyJWT

# 创建非 root 用户（安全加固）
RUN useradd -m -u 1000 litman && \
    chown -R litman:litman /app

USER litman

# 暴露端口
EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:8080/ || exit 1

# 启动命令
CMD ["litman", "--data-dir", "/data", "web", "serve", "--host", "0.0.0.0", "--port", "8080", "--auth-file", "/data/web_auth.json"]
```

#### 2.2 创建 `docker-compose.yml`
```yaml
version: '3.8'

services:
  app:
    build: .
    container_name: research-manager
    restart: unless-stopped
    ports:
      - "127.0.0.1:8080:8080"
    volumes:
      - ./data/local:/data
    environment:
      - TZ=Asia/Shanghai
      - PYTHONUNBUFFERED=1
    healthcheck:
      test: ["CMD", "curl", "-f", "http://localhost:8080/"]
      interval: 30s
      timeout: 10s
      retries: 3
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"

  nginx:
    image: nginx:alpine
    container_name: research-manager-nginx
    restart: unless-stopped
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx.conf:/etc/nginx/conf.d/default.conf:ro
      - ./ssl:/etc/nginx/ssl:ro
    depends_on:
      - app
    logging:
      driver: "json-file"
      options:
        max-size: "10m"
        max-file: "3"
```

#### 2.3 创建 `nginx.conf`
```nginx
upstream app_server {
    server app:8080;
}

server {
    listen 80;
    server_name your-domain.com;  # 修改为实际域名或 IP

    # 请求体大小限制（支持大文件上传）
    client_max_body_size 50M;

    # 日志
    access_log /var/log/nginx/research-manager-access.log;
    error_log /var/log/nginx/research-manager-error.log;

    location / {
        proxy_pass http://app_server;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # 超时设置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;

        # 安全头部
        add_header X-Frame-Options "SAMEORIGIN" always;
        add_header X-Content-Type-Options "nosniff" always;
        add_header X-XSS-Protection "1; mode=block" always;
    }

    # 限流配置（防止暴力破解）
    limit_req_zone $binary_remote_addr zone=login_limit:10m rate=5r/m;
    location /login {
        limit_req zone=login_limit burst=3 nodelay;
        proxy_pass http://app_server;
    }
}
```

#### 2.5 创建 `deploy.sh` 部署脚本
```bash
#!/bin/bash
set -e

echo "🚀 开始部署科研成果管理软件..."

# 1. 检查 Docker 是否安装
if ! command -v docker &> /dev/null; then
    echo "❌ Docker 未安装，请先安装 Docker"
    exit 1
fi

# 2. 检查 Docker Compose 是否安装
if ! command -v docker-compose &> /dev/null; then
    echo "❌ Docker Compose 未安装，请先安装 Docker Compose"
    exit 1
fi

# 3. 创建数据目录
mkdir -p data/local
mkdir -p ssl

# 4. 修改 nginx.conf 中的域名（提示用户手动修改）
echo "⚠️  请确认已修改 nginx.conf 中的 server_name 为实际域名或 IP"
read -p "是否已修改？(y/n) " -n 1 -r
echo
if [[ ! $REPLY =~ ^[Yy]$ ]]; then
    echo "❌ 请先修改 nginx.conf，然后重新运行脚本"
    exit 1
fi

# 5. 构建镜像
echo "📦 构建 Docker 镜像..."
docker-compose build

# 6. 启动服务
echo "🔧 启动服务..."
docker-compose up -d

# 7. 等待服务就绪
echo "⏳ 等待服务启动（最多等待 30 秒）..."
for i in {1..30}; do
    if curl -f http://localhost:8080 > /dev/null 2>&1; then
        echo "✅ 服务启动成功！"
        break
    fi
    if [ $i -eq 30 ]; then
        echo "❌ 服务启动超时，查看日志:"
        docker-compose logs app
        exit 1
    fi
    sleep 1
done

# 8. 显示服务状态
echo ""
echo "📊 服务状态:"
docker-compose ps

# 9. 显示访问信息
echo ""
echo "✅ 部署成功！"
echo "📍 本地访问: http://localhost"
echo "📍 外网访问: http://your-domain.com（需配置域名解析）"
echo "🔐 默认账号: admin / ChangeMe123"
echo ""
echo "⚠️  重要：请立即登录并修改默认密码！"
echo ""
echo "📝 常用命令:"
echo "  查看日志: docker-compose logs -f app"
echo "  停止服务: docker-compose down"
echo "  重启服务: docker-compose restart"
echo "  更新部署: git pull && docker-compose up -d --build"
```

```bash
docker-compose up -d

# 停止服务
docker-compose down

# 重启服务
docker-compose restart

# 查看日志
docker-compose logs -f app

# 查看容器状态
docker-compose ps

docker-compose exec app bash

git pull
docker-compose up -d --build

# 备份数据

tar -xzf backup-YYYYMMDD.tar.gz
```
